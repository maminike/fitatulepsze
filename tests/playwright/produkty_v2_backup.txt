"""
Playwright tests for the Products Catalog (/produkty) page
Tests cover search, filter, sort, add, and delete functionality
Total: 8 test cases
"""

import asyncio
import pytest
from playwright.async_api import Page


BASE_URL = "http://localhost:3000"


# Test 1
@pytest.mark.asyncio
async def test_1_load_produkty_page_and_verify_table_loads(page: Page):
    """Test 1: Load produkty page and verify that the product table loads"""
    await page.goto(f"{BASE_URL}/produkty")
    await page.wait_for_load_state("networkidle")

    # Verify page title contains "produkt" or "Katalog"
    title = await page.title()
    assert "produkt" in title.lower() or "katalog" in title.lower()

    # Verify table is visible with data
    table = page.locator("table")
    assert await table.is_visible()

    # Verify table has rows (header + at least one product)
    rows = page.locator("tbody tr")
    count = await rows.count()
    assert count > 0, "Product table should have at least one product"

# Test 5
@pytest.mark.asyncio
async def test_5_add_custom_product_form_submission(page: Page):
    """Test 5: Add a custom product using the form"""
    await page.goto(f"{BASE_URL}/produkty")
    await page.wait_for_load_state("networkidle")

    # Find "Add Product" button
    add_button = page.locator('[data-testid="add-product-button"]')
    if await add_button.is_visible():
        # Click to open add product modal/form
        await add_button.click()
        await asyncio.sleep(0.5)

        # Wait for form to appear
        form = page.locator("form")
        if await form.is_visible():
            # Fill product name
            name_input = page.locator('[data-testid="product-name-input"]')
            if await name_input.is_visible():
                await name_input.fill("Test Product XYZ")

            # Fill calories
            calories_input = page.locator('[data-testid="product-calories-input"]')
            if await calories_input.is_visible():
                await calories_input.fill("250")

            # Fill protein
            protein_input = page.locator('[data-testid="product-protein-input"]')
            if await protein_input.is_visible():
                await protein_input.fill("20")

            # Fill carbs
            carbs_input = page.locator('[data-testid="product-carbs-input"]')
            if await carbs_input.is_visible():
                await carbs_input.fill("30")

            # Fill fat
            fat_input = page.locator('[data-testid="product-fat-input"]')
            if await fat_input.is_visible():
                await fat_input.fill("8")

            # Submit form
            submit_button = page.locator('[data-testid="product-submit-button"]')
            if await submit_button.is_visible():
                await submit_button.click()
                await asyncio.sleep(1)
                await page.wait_for_load_state("networkidle")

# Test 8
@pytest.mark.asyncio
async def test_8_pagination_or_product_count_limit(page: Page):
    """Test 8: Verify products page handles multiple products properly"""
    await page.goto(f"{BASE_URL}/produkty")
    await page.wait_for_load_state("networkidle")

    # Get total row count
    rows = page.locator("tbody tr")
    row_count = await rows.count()

    # Verify reasonable number of products displayed
    assert row_count > 0, "Should display at least one product"
    assert row_count <= 200, "Should not display excessive products without pagination"

    # Verify all visible rows have expected data (cells)
    if row_count > 0:
        first_row = rows.first
        cells = first_row.locator("td")
        cell_count = await cells.count()
        assert cell_count > 0, "Each product row should have multiple cells"
